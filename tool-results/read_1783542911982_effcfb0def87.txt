   105→const fmtDate = (s: string | null) => {
   106→  if (!s) return '—';
   107→  try {
   108→    // Use 'en-GB' to get DD/MM/YYYY with English numerals (instead of Arabic-Indic ٠١٢٣)
   109→    return new Date(s).toLocaleDateString('en-GB', { year: 'numeric', month: '2-digit', day: '2-digit' });
   110→  } catch { return s; }
   111→};

   360→        ? detail.revisions[detail.revisions.length - 1].revNumber
   361→        : 0;
   362→      // Update the latest revision with the reply date, action, and approval type
   363→      const r = await fetch(`/api/transmittals/${id}/revisions`, {
   364→        method: 'POST',
   365→        headers: { 'Content-Type': 'application/json' },
   366→        body: JSON.stringify({
   367→          revNumber: latestRev,
   368→          replyDate: replyDate || null,
   369→          action: action || null,
   370→          approvalType: action === 'approved' ? (approvalType || null) : null,

   373→      });

   375→        const err = await r.json();
   376→        throw new Error(err.error || 'فشل تسجيل رد الاستشاري');
   377→      }
   378→      // Also update the consultant review status
   379→      // Map approval types to consultant statuses: NOT_APPROVED(D) → Submit Next Rev, FOR_INFORMATION(E) → Under Review
   380→      const actionToStatus: Record<string, string> = {
   381→        approved: (approvalType === 'NOT_APPROVED' ? 'Submit Next Rev' :
   382→                   approvalType === 'FOR_INFORMATION' ? 'Under Review' :
   383→                   approvalType === 'APPROVED_AS_NOTED_RESUBMIT' ? 'Submit Next Rev' : 'Approved'),
   384→        rejected: 'Submit Next Rev',
   385→        withdrawn: 'Cancelled',
   386→      };
   387→      const status = actionToStatus[action] || 'Under Review';
   388→      await fetch(`/api/transmittals/${id}/reviews`, {
   389→        method: 'POST',
   390→        headers: { 'Content-Type': 'application/json' },
   391→        body: JSON.stringify({ party: 'CONSULTANT', status, reviewDate: replyDate || null, notes: notes || null }),
   392→      });
   393→      const actionLabel = action === 'approved' && approvalType
   394→        ? `${getApprovalTypeLetter(approvalType)} - ${getApprovalTypeLabel(approvalType)}`
   395→        : action || '';

   398→      if (view === 'list') fetchList();
   399→      else fetchDashboard();
