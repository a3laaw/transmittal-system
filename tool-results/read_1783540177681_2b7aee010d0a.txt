   105→const fmtDate = (s: string | null) => {
   106→  if (!s) return '—';
   107→  try {
   108→    // Use 'en-GB' to get DD/MM/YYYY with English numerals (instead of Arabic-Indic ٠١٢٣)
   109→    return new Date(s).toLocaleDateString('en-GB', { year: 'numeric', month: '2-digit', day: '2-digit' });
   110→  } catch { return s; }
   111→};

   372→        rejected: 'Submit Next Rev',
   373→        withdrawn: 'Cancelled',
   374→      };

   376→      await fetch(`/api/transmittals/${id}/reviews`, {
   377→        method: 'POST',
   378→        headers: { 'Content-Type': 'application/json' },
   379→        body: JSON.stringify({ party: 'CONSULTANT', status, reviewDate: replyDate || null, notes: notes || null }),

   386→      if (view === 'list') fetchList();
